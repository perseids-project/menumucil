The Code

Include the necessary css and javascript

<link rel="stylesheet" href="menumucil/css/menumucil.css">
<script src="//ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="//ajax.googleapis.com/ajax/libs/jqueryui/1.10.3/jquery-ui.min.js"></script>
<script src="menumucil/menumucil.js"></script>

Set a semi transparent background-color to prettify things (optional)

<style type="text/css">
	.menumucil {
		background-color: rgba( 255, 255, 255, 0.9 );
	}
</style>

Create menumucil container

<div class="menumucil">[Menu Content Here]</div>

Start menumucil

<script type="text/javascript">
	$(document).ready( function() {
		$( '.menumucil' ).menumucil({cover:true});
	});
</script>
